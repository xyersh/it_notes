## Флаги конфигурации



| Операция | Название | Назначение                 |
| -------- | -------- | -------------------------- |
| \|       | OR       | Комбинирование флагов      |
| &        | AND      | Проверка наличия флага     |
| ^        | XOR      | Включение/выключение флага |
| &^       | AND NOT  | Очистить(снять) флаг       |
|          |          |                            |
|          |          |                            |


```go
type DumpFlags int 
const ( 
	DumpHeaders DumpFlags = 1 << iota // 1 
	DumpBodies // 2 
	DumpRequests // 4 
	DumpResponses // 8 
	DumpAuth // 16 
	DumpFilters // 32 
	// ... 
	)
	
	type ConfigFlags int
	cfg := EnableCompression | LogQueries
	
	
	
	// назначить новый флаг к уже имеющейся конфигурации
	cfg |= fs.DumpHeaders
	
	// снять флаг в имеющейся конфигурации
	cfg &^= fs.DumpHeaders // AND NOT
	
	// назначить/снять флаг в имеющейся конфигурации
	cfg ^= fs.DumpHeaders // XOR
	
	// Проверить, все ли из списка установлены в конфигурации
	if cfg & (fs.DumpHeaders | fs.DumpBodies) == (fs.DumpHeaders | fs.DumpBodies) { 
	...
	}
```
