GIT
	git
	git help commit
	git config --global user.name
	git config --global user.email
	git config --global core.autocrlf true
	git config --global core.safecrlf warn
	git config --global init.defaultBranch main
	git config format.pretty "%h %ad | %s%d [%an]"			- change a data format for log presentation 
 	git config log.date short								- change the date format to short variant


	git init
	git add .
	git commit -m <msg>
	git status
	git diff												- compares the working tree with the latest commit	
	git diff HEAD 											- compares the working tree with the latest commit of the current branch
	git diff 4620193 example.html 							- compares the example.html file in the working tree with the said file at the commit number 4620193
	git diff <commit_from> <commit_to>						- compare 


	
	git log													- command to see the commit history.
	git log <file_name>										- gets the commits only for a specified file
	git log --oneline										- command to see the commit history where each commit is a single line
		--since="5 minutes ago"
		--until="5 minutes ago"
		--author="Your Name"
		--all
		--format="%h %ad | %s%d [%an]" --date=short
			%h 												- is the abbreviated hash of the commit.
			%ad 											- is the commit date.
			| 												- is just a visual separator.
			%s 												- is the commit message.
			%d 												- represents commit decorations (e.g. branch heads or tags).
			%an 											- is the name of the author.
			--date=short 									- keeps the date format short.

	git tag													- lists all tags
	git tag 1.0												- Tags the latest commit as version 1.0
	git branch 												- command to list all branches.
	git switch -c <new_branch>								- create new branch and switch on it
	git switch <branch>										- switch on another branch
	git branch -d style										- delete the <style> branch now that we've merged it into <main>
	
	
	git restore . 											- restore all the files from working tree to its last committed state
	git restore --staged . 									- unstage the changes
	git restore --staged --worktree .						- restore both the staging area and the working tree
	
	
	git commit --amend -m "Added HTML tags to hello.html" 	- replace the last commit with the new one.  add the staged changes to the last commit
	git revert <hash_commit>								- adds a new commit that includes reverse changes. this command does not delete the commit
	git reset main											- Reset the current branch to the latest commit from main.	
	git reset --soft
	git reset --hard
